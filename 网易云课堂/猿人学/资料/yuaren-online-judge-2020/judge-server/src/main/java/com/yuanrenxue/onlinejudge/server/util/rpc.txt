 SekiroClient.start("sekiro.virjar.com",
                Constants.defaultNatServerPort,
                "virjar", "yaunren12")
                .registerHandler(new ActionHandler() {
                    @Override
                    public String action() {
                        return "query";
                    }

                    @Override
                    public void handleRequest(SekiroRequest sekiroRequest, final SekiroResponse sekiroResponse) {
                        final long id = sekiroRequest.getInt("id");
                        final Class<?> appClass = RposedHelpers.findClass("com.yuanrenxue.onlinejudge2020.OnlineJudgeApp", lpparam.classLoader);
                        new Handler(Looper.getMainLooper()).post(new Runnable() {
                            @Override
                            public void run() {
                                String sign = (String) RposedHelpers.callStaticMethod(appClass, "getSign1", id);
                                sekiroResponse.success(sign);
                            }
                        });

                    }
                });
